// Mat  library or EJML  zero-indexed libraries
var A = M("0.6 -0.8; 0.8 0.6")
var Ainv = inv(A)   // compute the inverse
var Atrans = A~   // transpose
var distance = norm2(Ainv-Atrans)
var dA = det(A)  // is 1


// Matrix  library, one indexed
var A1 = M1("0.6 -0.8; 0.8 0.6")
var Ainv1 = inv(A1)   // compute the inverse
var Atrans1 = A1~   // transpose
var distance1 = norm2(Ainv1-Atrans1)
var dA1 = det(A1)  // is 1
